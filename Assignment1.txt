const second_Greatest_Lowest = (arr_num) => {
  arr_num.sort(function(x,y){ return x-y; });
  let uniqueList = [arr_num[0]];
  let result = [];
  
  for(let j=1; j < arr_num.length; j++)
  {
    if(arr_num[j-1] !== arr_num[j])
    {
      uniqueList.push(arr_num[j]);
    }
  }
    result.push(uniqueList[1],uniqueList[uniqueList.length-2]);
  return result.join(',');
}
